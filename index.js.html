        /**

     * @property {Function} appendNode Add a node to the end of the list
     * @returns {ListNode} Return the new list
     */
    appendNode(value){
        let node = makeNode(value)
        if(!this.tail){
            this.head = this.tail = node
            return node
        }
        this.tail.next = node
        this.tail = node
        return node
    }
    /**
     * @property {Function} prependNode Add a node to the start of the list
     * @returns {ListNode} Return the new list
     */
    prependNode(value){
        let node = makeNode(value)
        if(!this.head){
            this.head = this.tail = node
            return node
        }
        node.next = this.head
        this.head = node
        return node
    }
    /**
     * @property {Function} removeHeadNode Remove the first node in the list
     * @returns {ListNode}
     */
     removeHeadNode(value){
        let node = makeNode(value)
        if(!this.head){
            this.head = this.tail = node
            return node
        }
        node.next = this.head
        this.head = node
        return node
    }
    /**
     * @property {Function} printList Print out values from a singly-linked list
     * @returns {void}
     */
    printList(){
        let current = this.head
        while(current){
            console.log(current.value)
            current = current.next
        }
    }
    /**
     * @property {Function} sizeList Return the size of a singly-linked list
     * @returns {number}
     */
    sizeList(){
        let count = 0
        let current = this.head
        while(current){
            count++
            current=current.next
        }
        return count
    }
}
